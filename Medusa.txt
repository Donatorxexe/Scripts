    pad.PaddingTop = UDim.new(0, guiCfg.cardPadding)

    return content

-- Switch tab function
local function switchTab(id)
    currentTab = id
    for tabId, pg in pairs(tabPages) do
        pg.scroll.Visible = (tabId == id)
    end
    -- Move indicator
    for i, def in ipairs(tabDefs) do
        if def.id == id then
            local targetY = 10 + (i-1) * 36
            twPlay(tabIndicator, 0.2, {Position = UDim2.new(0, 0, 0, targetY)})
            twPlay(tabGlow, 0.2, {Position = UDim2.new(0, 0, 0, targetY)})
        end
    end
end

local function mkToggle(parent, labelTxt, default, x, y, onChange)
    local lbl = Instance.new("TextLabel")
    lbl.Size = UDim2.new(0, 140, 0, guiCfg.toggleH)
    lbl.Position = UDim2.new(0, x, 0, y)
    lbl.BackgroundTransparency = 1
    lbl.Font = Enum.Font.GothamMedium
    lbl.TextSize = guiCfg.fontSize
    lbl.TextColor3 = guiCfg.textColor
    lbl.TextXAlignment = Enum.TextXAlignment.Left
    lbl.Text = labelTxt
    lbl.Parent = parent

    local track = Instance.new("TextButton")
    track.Size = UDim2.new(0, guiCfg.toggleW, 0, guiCfg.toggleH)
    track.Position = UDim2.new(1, -(guiCfg.toggleW + 8), 0, y)
    track.BackgroundColor3 = default and C.accent or Color3.fromRGB(45, 45, 60)
    track.BorderSizePixel = 0
    track.Text = ""
    track.AutoButtonColor = false
    track.Parent = parent
    Instance.new("UICorner", track).CornerRadius = UDim.new(1, 0)

    local knob = Instance.new("Frame")
    knob.Size = UDim2.new(0, guiCfg.toggleH - 4, 0, guiCfg.toggleH - 4)
    knob.Position = default and UDim2.new(1, -(guiCfg.toggleH - 2), 0.5, -(guiCfg.toggleH - 4)/2)
                            or UDim2.new(0, 2, 0.5, -(guiCfg.toggleH - 4)/2)
    knob.BackgroundColor3 = Color3.new(1, 1, 1)
    knob.BorderSizePixel = 0
    knob.ZIndex = 5
    knob.Parent = track
    Instance.new("UICorner", knob).CornerRadius = UDim.new(1, 0)

    local on = default
    track.MouseButton1Click:Connect(function()
        on = not on
        twPlay(track, 0.15, {BackgroundColor3 = on and C.accent or Color3.fromRGB(45, 45, 60)})
        twPlay(knob, 0.15, {Position = on 
            and UDim2.new(1, -(guiCfg.toggleH - 2), 0.5, -(guiCfg.toggleH - 4)/2)
            or UDim2.new(0, 2, 0.5, -(guiCfg.toggleH - 4)/2)})
        if onChange then onChange(on) end
    end)
    return track, knob
end

local function mkSlider(parent, labelTxt, initVal, minV, maxV, x, y, onChange, isFloat)
    local row = Instance.new("Frame")
    row.Size = UDim2.new(1, -16, 0, 40)
    row.Position = UDim2.new(0, x, 0, y)
    row.BackgroundTransparency = 1
    row.ClipsDescendants = false
    row.Parent = parent

    local lbl = Instance.new("TextLabel")
    lbl.Size = UDim2.new(1, 0, 0, 16)
    lbl.BackgroundTransparency = 1
    lbl.Font = Enum.Font.GothamMedium
    lbl.TextSize = guiCfg.fontSize
    lbl.TextColor3 = guiCfg.textColor
    lbl.TextXAlignment = Enum.TextXAlignment.Left
    lbl.Text = labelTxt
    lbl.Parent = row

    local valLbl = Instance.new("TextLabel")
    valLbl.Size = UDim2.new(0, 50, 0, 16)
    valLbl.Position = UDim2.new(1, -50, 0, 0)
    valLbl.BackgroundTransparency = 1
    valLbl.Font = Enum.Font.Code
    valLbl.TextSize = 11
    valLbl.TextColor3 = C.accent
    valLbl.TextXAlignment = Enum.TextXAlignment.Right
    if isFloat then
        valLbl.Text = string.format("%.2f", initVal)
    else
        valLbl.Text = tostring(math.floor(initVal))
    end
    valLbl.Parent = row

    local track = Instance.new("Frame")
    track.Size = UDim2.new(1, 0, 0, guiCfg.sliderH)
    track.Position = UDim2.new(0, 0, 0, 22)
    track.BackgroundColor3 = Color3.fromRGB(30, 30, 42)
    track.BorderSizePixel = 0
    track.ClipsDescendants = false
    track.Parent = row

    local pct = math.clamp((initVal - minV) / (maxV - minV), 0, 1)
    local fill = Instance.new("Frame")
    fill.Size = UDim2.new(pct, 0, 1, 0)
    fill.BackgroundColor3 = C.accent
    fill.BorderSizePixel = 0
    fill.Parent = track

    local knob = Instance.new("Frame")
    knob.Size = UDim2.new(0, 14, 0, 14)
    knob.AnchorPoint = Vector2.new(0.5, 0.5)
    knob.Position = UDim2.new(pct, 0, 0.5, 0)
    knob.BackgroundColor3 = Color3.new(1, 1, 1)
    knob.BorderSizePixel = 0
    knob.ZIndex = 10
    knob.Parent = track
    Instance.new("UICorner", knob).CornerRadius = UDim.new(1, 0)
    local ksk = Instance.new("UIStroke", knob)
    ksk.Color = C.accent
    ksk.Thickness = 2

    local isDragging = false

    local function applyPos(mouseX)
        local bx = track.AbsolutePosition.X
        local bw = track.AbsoluteSize.X
        if bw <= 0 then return end
        local p = math.clamp((mouseX - bx) / bw, 0, 1)
        local val
        if isFloat then
            val = minV + p * (maxV - minV)
            val = math.floor(val * 100) / 100
            valLbl.Text = string.format("%.2f", val)
        else
            val = math.floor(minV + p * (maxV - minV))
            valLbl.Text = tostring(val)
        end
        fill.Size = UDim2.new(p, 0, 1, 0)
        knob.Position = UDim2.new(p, 0, 0.5, 0)
        if onChange then onChange(val) end
    end

    track.InputBegan:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 then
            isDragging = true; applyPos(i.Position.X)
            twPlay(knob, 0.1, {Size = UDim2.new(0, 18, 0, 18)})
        end
    end)
    knob.InputBegan:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 then
            isDragging = true
            twPlay(knob, 0.1, {Size = UDim2.new(0, 18, 0, 18)})
        end
    end)
    UIS.InputChanged:Connect(function(i)
        if isDragging and i.UserInputType == Enum.UserInputType.MouseMovement then
            applyPos(i.Position.X)
        end
    end)
    UIS.InputEnded:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 then
            isDragging = false
            twPlay(knob, 0.1, {Size = UDim2.new(0, 14, 0, 14)})
        end
    end)

    return lbl, valLbl
end

local function refreshPlayers()

    for _, c in ipairs(plrCards) do pcall(function() c:Destroy() end) end
    plrCards = {}

    local order = 2
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= player then
            local card = Instance.new("Frame")
            card.Size = UDim2.new(1, 0, 0, 70)
            card.BackgroundColor3 = guiCfg.cardColor
            card.BorderSizePixel = 0
            card.LayoutOrder = order

            order = order + 1
            local csk = Instance.new("UIStroke", card)
            csk.Color = guiCfg.borderColor

-- Toggle feature helper
local function toggleFeature(key, name, color)
    st[key] = not st[key]
    local on = st[key]
    -- Post-toggle side effects
    if key == "esp" then
        if not on then -- clearESP will be called
        else lastESPRefresh = os.time() end
    elseif key == "aimbot" then
        if not on then obj.lockedTarget = nil; rmbDown = false end
    elseif key == "silentAim" then
        -- nothing extra needed
    elseif key == "triggerBot" then
        -- nothing extra needed
    end
    local icon = on and "✅" or "❌"
    notify(icon .. " " .. name .. " " .. (on and "ON" or "OFF"), on and color or C.error)
end

local actBtns = {}
local actDefs = {
    {key="esp",        txt="ESP",         color=C.success},
    {key="aimbot",     txt="AIMBOT",      color=C.aimbot},

                if hrp then
                    obj.bv = Instance.new("BodyVelocity"); obj.bv.MaxForce = Vector3.new(1e5,1e5,1e5); obj.bv.Velocity = Vector3.zero; obj.bv.Parent = hrp
                    obj.bg = Instance.new("BodyGyro"); obj.bg.MaxTorque = Vector3.new(1e5,1e5,1e5); obj.bg.P = 1e4; obj.bg.Parent = hrp
                end
            else
                if obj.bv then pcall(function() obj.bv:Destroy() end); obj.bv = nil end
                if obj.bg then pcall(function() obj.bg:Destroy() end); obj.bg = nil end
            end
        elseif def.key == "hitbox" then
            if not st.hitbox then -- resetAllHitboxes called later
            end
        elseif def.key == "speed" then
            local hum = player.Character and player.Character:FindFirstChild("Humanoid")
            if hum then hum.WalkSpeed = st.speed and cfg.walkSpeed or 16 end

            if st.fullbright then
                Lighting.Brightness = 2
                Lighting.ClockTime = 14
                Lighting.FogEnd = 100000
                Lighting.FogStart = 100000
                Lighting.GlobalShadows = false

                    if eff:IsA("Atmosphere") then eff.Density = 0 end
                    if eff:IsA("BloomEffect") then eff.Enabled = false end
                    if eff:IsA("ColorCorrectionEffect") then eff.Enabled = false end
                end

                pcall(function()
                    Lighting.Brightness = obj.origLighting.Brightness or 1
                    Lighting.ClockTime = obj.origLighting.ClockTime or 14
                    Lighting.FogEnd = obj.origLighting.FogEnd or 10000
                    Lighting.FogStart = obj.origLighting.FogStart or 0

                            if eff:IsA("Atmosphere") then eff.Density = 0.3 end
                        end
                    end
                end)
            end
        elseif def.key == "esp" then
            if not st.esp then -- clearESP handled by esp loop
            end
        end
    end)
    actBtns[def.key] = {btn = btn, sk = sk, bar = bar, dot = dot, color = def.color}
end

        if hum then
            hum:SetStateEnabled(Enum.HumanoidStateType.FallingDown, false)
            hum:SetStateEnabled(Enum.HumanoidStateType.Ragdoll, false)
            -- Keep velocity safe on landing
            local hrp = player.Character:FindFirstChild("HumanoidRootPart")

                if vel.Y < -150 then

                end
            end
        end
    end
end)

local function removeESP(plr)
    local d = obj.espObjs[plr]
    if not d then return end
    pcall(function() if d.hl then d.hl:Destroy() end end)
    pcall(function() if d.bb then d.bb:Destroy() end end)
    pcall(function() if d.cn then d.cn:Disconnect() end end)
    obj.espObjs[plr] = nil
end

local function clearESP()
    for p in pairs(obj.espObjs) do removeESP(p) end
end

    if not st.esp or not plr or plr == player or obj.espObjs[plr] then return end
    if cfg.teamCheck and plr.Team and player.Team and plr.Team == player.Team then return end
    local char = plr.Character
    if not char then return end
    local head = char:FindFirstChild("Head")
    if not head then return end

    -- Distance check for ESP
    local hrpCheck = char:FindFirstChild("HumanoidRootPart")

        local dist = (hrpCheck.Position - camera.CFrame.Position).Magnitude
        if dist > cfg.espMaxDist then return end
    end

    local hl

    hl.FillColor = C.accent
    hl.OutlineColor = C.accent:Lerp(Color3.new(0,0,0), 0.5)
    hl.Adornee = char
    hl.Parent = char

    local bb = Instance.new("BillboardGui")
    bb.Name = "ESP_BB_" .. plr.Name
    bb.Adornee = head
    bb.AlwaysOnTop = true
    bb.Size = UDim2.new(0, 140, 0, 50)
    bb.StudsOffset = Vector3.new(0, 3.5, 0)
    bb.Parent = CoreGui

    nl.BackgroundTransparency = 1

    nl.TextStrokeTransparency = 0.4
    nl.Text = plr.DisplayName
    nl.Parent = bb

    dl.Size = UDim2.new(1, 0, 0, 14)
    dl.Position = UDim2.new(0, 0, 0, 18)
    dl.BackgroundTransparency = 1

    dl.TextSize = 12
    dl.TextColor3 = C.accent

    local hpBg = Instance.new("Frame")
    hpBg.Size = UDim2.new(0.8, 0, 0, 4)

    hpBg.BackgroundColor3 = Color3.fromRGB(30, 30, 42)

local function isValidTarget(plr)
    if not plr or plr == player or not plr.Character or not plr.Character.Parent then return false end
    if cfg.teamCheck and plr.Team and player.Team and plr.Team == player.Team then return false end
    local hum = plr.Character:FindFirstChild("Humanoid")
    if not hum or hum.Health <= 0 then return false end
    -- Health Check
    if cfg.healthCheck then
        local hpPct = (hum.Health / hum.MaxHealth) * 100
        if hpPct < cfg.healthMin then return false end
    end
    -- Distance check
    local hrp = plr.Character:FindFirstChild("HumanoidRootPart")
    if not hrp then return false end
    local dist = (hrp.Position - camera.CFrame.Position).Magnitude
    if dist > cfg.maxDistance then return false end
    return true
end

local function getAimPart(char)
    if not char then return nil end
    if cfg.aimbotPart == "Head" then
        return char:FindFirstChild("Head")
    elseif cfg.aimbotPart == "Torso" then
        return char:FindFirstChild("UpperTorso") or char:FindFirstChild("Torso")
    elseif cfg.aimbotPart == "Closest" then
        local mp = UIS:GetMouseLocation()
        local best, bestD = nil, math.huge
        for _, nm in ipairs({"Head","UpperTorso","Torso","LowerTorso","HumanoidRootPart","LeftUpperArm","RightUpperArm","LeftUpperLeg","RightUpperLeg"}) do
            local part = char:FindFirstChild(nm)
            if part and part:IsA("BasePart") then
                local sp, on = camera:WorldToViewportPoint(part.Position)
                if on then
                    local d = (Vector2.new(sp.X, sp.Y) - mp).Magnitude
                    if d < bestD then bestD = d; best = part end
                end
            end
        end
        return best
    end
    return char:FindFirstChild("Head")
end

local function checkVisible(part)
    if not cfg.visibleCheck then return true end
    local origin = camera.CFrame.Position
    local dir = (part.Position - origin)
    local rayParams = RaycastParams.new()
    rayParams.FilterType = Enum.RaycastFilterType.Exclude
    rayParams.FilterDescendantsInstances = {player.Character, camera}
    local result = WS:Raycast(origin, dir, rayParams)
    if result then
        return result.Instance:IsDescendantOf(part.Parent)
    end
    return true
end

local function closestInFOV()
    local mp = UIS:GetMouseLocation()
    local best, bestD = nil, math.huge
    for _, plr in ipairs(Players:GetPlayers()) do
        if isValidTarget(plr) then
            local part = getAimPart(plr.Character)
            if part then
                local sp, on = camera:WorldToViewportPoint(part.Position)
                if on then
                    local d = (Vector2.new(sp.X, sp.Y) - mp).Magnitude
                    if d < cfg.aimbotFOV and d < bestD then
                        if checkVisible(part) then
                            best = plr; bestD = d
                        end
                    end
                end
            end
        end
    end
    return best
end

        if obj.fovCircle then
            obj.fovCircle.Position = mp

        if not obj.lockedTarget then
            obj.lockedTarget = closestInFOV()

    else

        -- Only clear target if silent aim and trigger bot aren't using it
        if not (st.silentAim or st.triggerBot) then
            obj.lockedTarget = nil
        end
    end

    -- Aimbot camera pull
    if st.aimbot and rmbDown and obj.lockedTarget and obj.lockedTarget.Character then
        local part = getAimPart(obj.lockedTarget.Character)
        if part then
            local targetPos = part.Position
            if cfg.aimSmooth == 0 then
                camera.CFrame = CFrame.new(camera.CFrame.Position, targetPos)
            else
                local t = (1 - cfg.aimSmooth / 100) * 0.93 + 0.02
                local currentDir = camera.CFrame.LookVector
                local targetDir = (targetPos - camera.CFrame.Position).Unit
                local blended = currentDir:Lerp(targetDir, t).Unit
                camera.CFrame = CFrame.new(camera.CFrame.Position, camera.CFrame.Position + blended)

        end

    for key, pill in pairs(stPills) do

        for _, def in ipairs(pillDefs) do

        if hum and part then

        end
    else
        lockLbl.Text = "🔓 No target locked"
        lockLbl.TextColor3 = guiCfg.mutedColor
        lockCard.BackgroundColor3 = guiCfg.cardColor
    end

    -- Info card
    infoLbl.Text = "FOV: "..cfg.aimbotFOV.." | Smooth: "..cfg.aimSmooth.." | Part: "..cfg.aimbotPart
        .."\nDist: "..cfg.maxDistance.." | HP Check: "..(cfg.healthCheck and ">"..cfg.healthMin.."%" or "OFF")

    if st.esp and st.autoRefresh then
        local rem = cfg.espRefreshInterval - (os.time() - lastESPRefresh)
        refLbl.Text = string.format("ESP Refresh: %d:%02d", math.floor(rem/60), rem%60)
    else
        refLbl.Text = "ESP Refresh: " .. (st.esp and "Manual" or "OFF")

                            return oldNamecall(self, unpack(args))

                    elseif method == "Raycast" then
                        local args = {...}
                        if typeof(args[1]) == "Vector3" then
                            local origin = args[1]
                            args[2] = (part.Position - origin).Unit * 1000
                            return oldNamecall(self, unpack(args))
                        end
                    end
                end
            end
            return oldNamecall(self, ...)
        end)
    end
end)

local crossGui = Instance.new("ScreenGui")
crossGui.Name = "MedusaCross"

crossGui.DisplayOrder = 1002
crossGui.IgnoreGuiInset = true

local crossFrame = Instance.new("Frame")
crossFrame.Size = UDim2.new(1, 0, 1, 0)
crossFrame.BackgroundTransparency = 1
crossFrame.Parent = crossGui

local function updateCrosshair()
    for _, c in ipairs(crossFrame:GetChildren()) do c:Destroy() end
    if not st.crosshair then return end

    local vp = camera.ViewportSize
    local cx, cy = vp.X / 2, vp.Y / 2
    local s = cfg.crosshairSize
    local g = cfg.crosshairGap
    local t = cfg.crosshairThick

    local function mkLine(x, y, w, h)
        local l = Instance.new("Frame")
        l.Size = UDim2.new(0, w, 0, h)
        l.Position = UDim2.new(0, x, 0, y)
        l.BackgroundColor3 = C.accent
        l.BorderSizePixel = 0
        l.Parent = crossFrame
        local lsk = Instance.new("UIStroke", l)
        lsk.Color = Color3.new(0, 0, 0)
        lsk.Thickness = 1
        return l
    end

    local style = cfg.crosshairStyle
    if style == "Cross" then
        mkLine(cx - s - g, cy - t/2, s, t)
        mkLine(cx + g, cy - t/2, s, t)
        mkLine(cx - t/2, cy - s - g, t, s)
        mkLine(cx - t/2, cy + g, t, s)
    elseif style == "Dot" then
        mkLine(cx - t, cy - t, t*2, t*2)
    elseif style == "Circle" then
        local circle = Instance.new("Frame")
        circle.Size = UDim2.new(0, s*2, 0, s*2)
        circle.Position = UDim2.new(0, cx - s, 0, cy - s)
        circle.BackgroundTransparency = 1
        circle.Parent = crossFrame

UIS.InputBegan:Connect(function(i, gp)
    if not st.running then return end

    -- Keybind assignment mode
    if bindWaiting and i.KeyCode ~= Enum.KeyCode.Unknown then
        keybinds[bindWaiting] = i.KeyCode
        if bindBtns[bindWaiting] then
            bindBtns[bindWaiting].Text = i.KeyCode.Name
            bindBtns[bindWaiting].TextColor3 = C.accent
            local bsk = bindBtns[bindWaiting]:FindFirstChildWhichIsA("UIStroke")
            if bsk then bsk.Color = guiCfg.borderColor end
        end
        notify("🎮 " .. bindWaiting .. " → " .. i.KeyCode.Name, C.accent)
        bindWaiting = nil
        return
    end

    -- RMB for aimbot
    if i.UserInputType == Enum.UserInputType.MouseButton2 then
        if st.aimbot then rmbDown = true end
        return
    end

    if gp then return end
    if i.KeyCode == Enum.KeyCode.Unknown then return end
    local k = i.KeyCode

    -- Process keybinds
    if k == keybinds.esp then
        toggleFeature("esp", "ESP", C.success)
        if not st.esp then clearESP() else lastESPRefresh = os.time() end
    elseif k == keybinds.aimbot then
        toggleFeature("aimbot", "Aimbot", C.aimbot)
        if not st.aimbot then obj.lockedTarget = nil; rmbDown = false end
    elseif k == keybinds.silentAim then
        toggleFeature("silentAim", "Silent Aim", C.aimbot)
    elseif k == keybinds.triggerBot then
        toggleFeature("triggerBot", "Trigger Bot", C.warning)
    elseif k == keybinds.fly then
        toggleFeature("fly", "Fly", C.fly)
        if st.fly then
            local hrp = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
            if hrp then
                obj.bv = Instance.new("BodyVelocity"); obj.bv.MaxForce = Vector3.new(1e5,1e5,1e5); obj.bv.Velocity = Vector3.zero; obj.bv.Parent = hrp
                obj.bg = Instance.new("BodyGyro"); obj.bg.MaxTorque = Vector3.new(1e5,1e5,1e5); obj.bg.P = 1e4; obj.bg.Parent = hrp
            end
        else
            if obj.bv then pcall(function() obj.bv:Destroy() end); obj.bv = nil end
            if obj.bg then pcall(function() obj.bg:Destroy() end); obj.bg = nil end
        end
    elseif k == keybinds.hitbox then
        toggleFeature("hitbox", "Hitbox", C.warning)
        if not st.hitbox then resetAllHitboxes() end
    elseif k == keybinds.noclip then
        toggleFeature("noclip", "Noclip", C.success)
    elseif k == keybinds.speed then
        st.speed = not st.speed
        local hum = player.Character and player.Character:FindFirstChild("Humanoid")
        if hum then hum.WalkSpeed = st.speed and cfg.walkSpeed or 16 end
        notify(st.speed and "🏃 Speed ON" or "❌ Speed OFF", st.speed and C.success or C.error)
    elseif k == keybinds.infJump then
        st.infJump = not st.infJump
        notify(st.infJump and "🦘 Inf Jump ON" or "❌ Inf Jump OFF", st.infJump and C.success or C.error)
    elseif k == keybinds.fullbright then
        toggleFeature("fullbright", "Fullbright", C.warning)
        if st.fullbright then
            Lighting.Brightness = 2; Lighting.ClockTime = 14; Lighting.FogEnd = 100000; Lighting.FogStart = 100000; Lighting.GlobalShadows = false
            for _, eff in ipairs(Lighting:GetChildren()) do
                if eff:IsA("Atmosphere") then pcall(function() eff.Density = 0 end) end
                if eff:IsA("BloomEffect") or eff:IsA("ColorCorrectionEffect") then pcall(function() eff.Enabled = false end) end
            end
        else
            pcall(function()
                Lighting.Brightness = obj.origLighting.Brightness or 1
                Lighting.FogEnd = obj.origLighting.FogEnd or 10000
                Lighting.FogStart = obj.origLighting.FogStart or 0
                Lighting.GlobalShadows = obj.origLighting.GlobalShadows ~= false
                for _, eff in ipairs(Lighting:GetChildren()) do
                    if eff:IsA("Atmosphere") or eff:IsA("BloomEffect") or eff:IsA("ColorCorrectionEffect") then
                        eff.Enabled = true
                    end
                end
            end)
        end
    elseif k == keybinds.clickTp then
        st.clickTp = not st.clickTp
        notify(st.clickTp and "🖱️ Click TP ON" or "❌ Click TP OFF", st.clickTp and C.success or C.error)
    elseif k == keybinds.noFallDmg then
        st.noFallDmg = not st.noFallDmg
        notify(st.noFallDmg and "🛡️ No Fall Dmg ON" or "❌ No Fall Dmg OFF", st.noFallDmg and C.success or C.error)
    elseif k == keybinds.tpCursor then
        -- TP to mouse cursor position
        local hrp = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
        if hrp and mouse.Hit then
            hrp.CFrame = mouse.Hit + Vector3.new(0, 3, 0)
            notify("🏃 TP to cursor!", C.success)
        end
    elseif k == keybinds.toggleGui then
        st.gui = not st.gui
        panel.Visible = st.gui
        shadow.Visible = st.gui
        notify(st.gui and "👁️ GUI Visible" or "❌ GUI Hidden", C.accent)
    elseif k == keybinds.eject then
        notify("⚠️ Ejecting...", C.error)
        task.delay(0.5, function() if eject then eject() end end)
    elseif k == keybinds.panic then
        -- PANIC — disable everything
        for key2 in pairs(st) do
            if key2 ~= "gui" and key2 ~= "running" and key2 ~= "minimized" and key2 ~= "antiAfk" and key2 ~= "autoRefresh" then
                st[key2] = false
            end
        end
        rmbDown = false; obj.lockedTarget = nil
        if obj.bv then pcall(function() obj.bv:Destroy() end); obj.bv = nil end
        if obj.bg then pcall(function() obj.bg:Destroy() end); obj.bg = nil end
        clearESP(); resetAllHitboxes()
        local hum = player.Character and player.Character:FindFirstChild("Humanoid")
        if hum then hum.WalkSpeed = 16 end
        -- Restore lighting
        pcall(function()
            Lighting.Brightness = obj.origLighting.Brightness or 1
            Lighting.FogEnd = obj.origLighting.FogEnd or 10000
            Lighting.GlobalShadows = obj.origLighting.GlobalShadows ~= false
        end)
        notify("🚨 PANIC — Everything disabled!", C.warning)
    end
end)

eject = function()
    print("[MEDUSA] Ejecting...")
    st.running = false
    for k in pairs(st) do st[k] = false end
    rmbDown = false

    clearESP(); resetAllHitboxes()
    if obj.bv then pcall(function() obj.bv:Destroy() end); obj.bv = nil end
    if obj.bg then pcall(function() obj.bg:Destroy() end); obj.bg = nil end
    pcall(function() if obj.fovCircle then obj.fovCircle:Remove(); obj.fovCircle = nil end end)
    obj.lockedTarget = nil; obj.spectating = nil

    -- Reset camera
    pcall(function()
        camera.CameraSubject = player.Character and player.Character:FindFirstChild("Humanoid")
    end)

    -- Reset character
    local hum = player.Character and player.Character:FindFirstChild("Humanoid")
    if hum then hum.WalkSpeed = 16 end
    local char = player.Character
    if char then
        for _, p in ipairs(char:GetDescendants()) do
            if p:IsA("BasePart") then p.CanCollide = true end
        end
        local hrp = char:FindFirstChild("HumanoidRootPart")
        if hrp then
            for _, o in pairs(hrp:GetChildren()) do
                if o:IsA("BodyVelocity") or o:IsA("BodyGyro") then o:Destroy() end
            end
        end
    end

    -- Restore lighting
    pcall(function()
        Lighting.Brightness = obj.origLighting.Brightness or 1
        Lighting.ClockTime = obj.origLighting.ClockTime or 14
        Lighting.FogEnd = obj.origLighting.FogEnd or 10000
        Lighting.FogStart = obj.origLighting.FogStart or 0
        Lighting.GlobalShadows = obj.origLighting.GlobalShadows ~= false
        for _, eff in ipairs(Lighting:GetChildren()) do
            if eff:IsA("Atmosphere") or eff:IsA("BloomEffect") or eff:IsA("ColorCorrectionEffect") then
                eff.Enabled = true
                if eff:IsA("Atmosphere") then pcall(function() eff.Density = 0.3 end) end
            end
        end
    end)

    -- Destroy GUIs
    pcall(function() screenGui:Destroy() end)
    pcall(function() crossGui:Destroy() end)
    pcall(function() wmGui:Destroy() end)
    pcall(function() thGui:Destroy() end)
    for _, g in pairs(playerGui:GetChildren()) do
        if g.Name == "MedusaNotifs" or g.Name == "MedusaGUI" then g:Destroy() end
    end
    for _, g in pairs(CoreGui:GetChildren()) do
        if g.Name and g.Name:match("^ESP_") then g:Destroy() end
    end

    obj.espObjs = {}
    obj.origSizes = {}
    print("[MEDUSA] Ejected successfully!")
end
